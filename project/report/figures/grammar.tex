\begin{grammar}

<Program> ::= (<Statement> | <Block>)* <Expression>?

<Statement> ::= <VarDecl>
              \alt <FunctionDecl>
              \alt <ReturnStmt>
              \alt <PrintStmt>
              \alt <BreakStmt>
              \alt <ExpressionStmt>

<VarDecl> ::= "var" IDENTIFIER ("=" <Expression>)? ";"

<FunctionDecl> ::= "fn" IDENTIFIER "(" <Parameters>? ")" <Block>

<Parameters> ::= IDENTIFIER ("," IDENTIFIER)*

<ReturnStmt> ::= "return" <Expression> ";"

<PrintStmt> ::= "print" <Expression> ";"

<ExpressionStmt> ::= <Expression> ";"

<Expression> ::= <WhileExpr>
               \alt <LoopExpr>
               \alt <IfExpr>
               \alt <Block>
               \alt <Assignment>

<WhileExpr> ::= "while" <Expression> (":" IDENTIFIER)? <Block>

<LoopExpr> ::= "loop" (":" IDENTIFIER)? <Block>

<IfExpr> ::= "if" <Expression> <Expression> ("else" <Expression>)?

<BreakStmt> ::= "break" (: IDENTIFIER)? ("return" <Expression>)? ";"

<Block> ::= "{" (<Statement> | <Block>)* <Expression>? "}"

<Assignment> ::= IDENTIFIER "=" <Expression>
               \alt <LogicOr>

<LogicOr> ::= <LogicAnd> "or" <LogicOr>
            \alt <LogicAnd>

<LogicAnd> ::= <Equality> "and" <LogicAnd>
            \alt <Equality>

<Equality> ::= <Comparison> ("==" | "!=") <Equality>
            \alt <Comparison>

<Comparison> ::= <Term> ("<" | "<=" | ">" | ">=") <Comparison>
            \alt <Term>

<Term> ::= <Factor> ("+" | "-") <Term>
            \alt <Factor>

<Factor> ::= <Exponent> ("*" | "/") <Factor>
            \alt <Exponent>

<Exponent> ::= <Unary> "**" <Exponent>
            \alt <Unary>

<Unary> ::= ("-" | "!") <Unary>
          \alt <Call>

<Call> ::= <Primary> ("(" <Arguments>? ")")*

<Arguments> ::= <Expression> ("," <Expression>)*

<Primary> ::= NUMBER
            \alt STRING
            \alt "true"
            \alt "false"
            \alt "null"
            \alt IDENTIFIER
            \alt "(" <Expression> ")"


\end{grammar}
